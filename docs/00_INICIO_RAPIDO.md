function App() {

          <Route path="/projects" element={<ProjectList />} />

function Dashboard() {

function StatCard({ title, value, color }) {

function CreateProject() {

  const [formData, setFormData] = useState({
    name: '',
    clientName: '',
    location: '',
  });

  const handleSubmit = (e) => {

    <div>
      <h2>Criar Novo Projeto</h2>

          onClick={() => setProjectType('decor')}

            borderRadius: '4px',
            cursor: 'pointer',

        >
          Simu

        <div style={{ marginBottom: '15px' }}>
          <label style={{ display: 'block', marginBottom: '5px' }}>Nome do Projeto *</label>
          <input
            type="text"
            required
            value={formData.name}
            onChange={(e) => setFormData({...formData, name: e.target.value})}
            style={{ width: '100%', padding: '8px', borderRadius: '4px', border: '1px solid #ddd' }}
          />
        </div>

        <div style={{ marginBottom: '15px' }}>
          <label style={{ display: 'block', marginBottom: '5px' }}>Cliente *</label>
          <input
            type="text"
            required
            value={formData.clientName}
            onChange={(e) => setFormData({...formData, clientName: e.target.value})}
            style={{ width: '100%', padding: '8px', borderRadius: '4px', border: '1px solid #ddd' }}
          />
        </div>

        <div style={{ marginBottom: '15px' }}>
          <label style={{ display: 'block', marginBottom: '5px' }}>Localiza√ß√£o</label>
          <input
            type="text"
            value={formData.location}
            onChange={(e) => setFormData({...formData, location: e.target.value})}

          type="submit"
          style={{
            padding: '10px 20px',
            background: '#2196F3',
            color: '#fff',

          }}

    </div>
  );
}

export default App;
